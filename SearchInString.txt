class NEW {
    public static void main(String[] args) {
        String str="DEEPU";
        char target='P';
        System.out.println(SerachInString(str,target));
    }

    static boolean SerachInString(String str, char target){
        if(str.length()==0){
            return false;
        }
        for(int i=0; i<str.length(); i++){
            if(target==str.charAt(i)){
                return true;
            }
        }
        return false;

    }


}
