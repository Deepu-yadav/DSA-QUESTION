import java.util.Arrays;

// In bubble sort we compare the adjacent element and it also called implace algorithm

public class BubbleSort {
    public static void main(String[] args) {
        int []arr={4,5,6,2,3,1,9};
        Sorting(arr);
        System.out.println(Arrays.toString(arr));

    }
    static void Sorting(int[]arr){
        boolean IsSwapped;
        for(int i=0; i<arr.length; i++){
            IsSwapped=false;
            
            // for each step, max element come at the last respective index
            for(int j=1; j<arr.length-i; j++){
                if(arr[j]<arr[j-1]){
                    // swapped
                    int temp=arr[j];
                    arr[j]=arr[j-1];
                    arr[j-1]=temp;
                    IsSwapped=true;
                }
            }
            // if you did not swap for a particular value of i , it means the array is sorted hence stop
            
            if(!IsSwapped){
                break;
            }
        }

    }
}
