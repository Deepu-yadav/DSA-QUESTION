import java.util.Arrays;
// Question: Search in the mountain array

class NEW {
    public static void main(String[] args) {
        int[] arr = {1, 2,3,4,5,3,1};
        int target=3;
        System.out.println(ans(arr,target));
    }

    static int ans(int[]arr,int target){
        int peak=-1;
         peak=PeakIndex(arr);

        if(arr[peak]==target){
            return peak;
        }
        if(target<arr[peak]){
            return OderAgnosticBinarySearch(arr,target,0,peak);
        }
        else {
            return OderAgnosticBinarySearch(arr,target,peak+1,arr.length-1);
        }
    }

    static int PeakIndex(int[]arr){

        int start=0;
        int end=arr.length-1;
        while (end>start){
            int mid=start+(end-start)/2;
            if(arr[mid]>arr[mid+1]){
                // its means we are in descending order
                end=mid;
            }
            else {
                start=mid+1;
            }
        }
        return start;  // here we can return either start or end
    }

    static int OderAgnosticBinarySearch(int[] arr,int target,int start,int end){
        // find weather the array is sorted in assccending OR descending
        boolean isAsync=arr[start]<arr[end];

        while (start<=end){

            int mid=start+(end-start)/2;

            if(arr[mid]==target){
                return mid;
            }


            if(isAsync){  // if it is execute that means array is in asccending order
                if(target>arr[mid]){
                    start=mid+1;
                }
                if(target<arr[mid]){
                    end=mid-1;
                }
            }
            else {  // if it is execute that means array is in descending array
                if(target>arr[mid]){
                    end=mid-1;
                }
                else {
                    start=mid+1;
                }
            }

        }
        return -1;
    }
