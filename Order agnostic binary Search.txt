/ WHEN We dont know in binary Search array is in asscending order or dexcending order then we will apply order agnostic binary Search

public class NEW {

    public static void main(String[] args) {
        int [] arr={-18,-12,-2,0,2,3,4,15,16,18,22,45,81};
        int target=81;
        System.out.println(Ans(arr,target));

    }
    static int Ans(int []arr,int target){

        int start=0;
        int end=arr.length-1;
        boolean isAsync=arr[start]<arr[end];

        while (start<=end){
            int mid=start+(end-start)/2;
            if(arr[mid]==target){
                return mid;
            }
            if(isAsync){  // if it is execute that means array is in asccending order
                if(target>arr[mid]){
                    start=mid+1;
                }
                if(target<arr[mid]){
                    end=mid-1;
                }
            }
            else {  // if it is execute that means array is in descending array
                if(target>arr[mid]){
                    end=mid-1;
                }
                else {
                    start=mid+1;
                }
            }

        }
        return -1;
    }

}
