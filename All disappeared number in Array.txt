import java.util.ArrayList;
import java.util.Arrays;
// Question: Find All dissappear number in the array

public class MisingNumber {
    public static void main(String[] args) {
        int []arr={4,3,2,7,8,2,3,1};
      //  ArrayList<Integer> ans = Sorting(arr);
       // System.out.println(Arrays.toString(ans));

        System.out.println(Sorting(arr));

    }
    static ArrayList<Integer> Sorting(int[]arr){
        ArrayList<Integer> list=new ArrayList<>();
        int i=0;
        while (i<arr.length){
            int CorrectIndex=arr[i]-1;
            if( arr[i]!=arr[CorrectIndex]){
                swap(arr,i,CorrectIndex);
            }
            else {
                i++;
            }
        }
        // Just find missing number

        for(i=0; i<arr.length; i++){
            if(arr[i]!=i+1){
                list.add(i+1);
               // return i+1;
            }
        }
        return list;

    }

    static void swap(int[] arr, int first, int second) {
        int temp=arr[first];
        arr[first]=arr[second];
        arr[second]=temp;

    }
}
