import java.util.*;
// Question Search a MIN Number in a array

class NEW {

    public static void main(String[] args) {
        int arr[]={18,12,0,14,-28};

         Min(arr);

    }

    static void Min(int[]arr){
        int min=arr[0];

        for(int i=0; i<arr.length; i++){
            if(min>arr[i]){
                min=arr[i];

            }
        }
        System.out.println(min);

    }
